Explanation of parameters


\begin{itemize}

\item Fleishman~\cite{fleishman2003bilateral}
\begin{itemize}
\item[-] $k_{iter}$: number of iterations.
\end{itemize}

\item Jones~\cite{jones2003non}
\begin{itemize}
\item[-] $\sigma_f/\|e\|$, $\sigma_g/\|e\|$: parameters that determine the variance for spatial and range kernels, with $\|e\|$ being the average edge length.
\end{itemize}

\item Sun~\cite{sun2007fast}
\begin{itemize}
\item[-] $T$: threshold for controlling the averaging weights (see Equation (8) of Sun~\cite{sun2007fast}).
\item[-] $k_{iter}$: number of iterations for updating normals.
\item[-] $v_{iter}$: number of iterations for a vertex update.
\end{itemize}

\item Local scheme of Zheng~\cite{zheng2011bilateral}
\begin{itemize}
\item[-] $\sigma_s$: variance parameter for the spatial kernel.
\item[-] $k_{iter}$: number of iterations for updating normals.
\item[-] $v_{iter}$: number of iterations for a vertex update.
\end{itemize}

\item He~\cite{he2013mesh}
\begin{itemize}
\item[-] $\lambda$: weight for the $L_0$ term in the target function.
\item[-] $\alpha_0, \beta_0$: initial values for $\alpha$ and $\beta$ in Algorithm 1 of He~\cite{he2013mesh}.
\item[-] $\mu_\alpha, \mu$: update ratios for $\alpha$ and $\beta$.
\item[-] $\beta_{max}$: maximum value of $\beta$.
\end{itemize}

\item Zhang~\cite{Zhang2015Filter}
\begin{itemize}
\item[-] $r$: radius for the geometrical neighborhood, also shown as the ratio with respect to the average
distance between neighboring face barycenter; not applicable if a topological neighborhood is
used.
\item[-] $\sigma_r$: variance of the range kernel.
\item[-] $k_{iter}$: number of iterations for updating normals.
\item[-] $v_{iter}$: number of iterations for a vertex update.
\end{itemize}


\item ours
\begin{itemize}
\item[-] $r$: radius for the geometrical neighborhood, also shown as the ratio with respect to the average
distance between neighboring face barycenter; 
\item[-] $\sigma_r$ and $\sigma_s$: variances of two kind of accumulative different along the geodesic path.
\item[-] $k_{iter}$: number of iterations for updating normals.
\item[-] $v_{iter}$: number of iterations for a vertex update.
\end{itemize}


\end{itemize}
